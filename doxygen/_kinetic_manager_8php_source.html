174  public function resolveClass(KineticNode $node, ?string $fqn, ?string $super) : ?object{
175  if($fqn === null){
176  return null;
177  }
178 
179  if($fqn{0} === '$'){
180  $object = $this->adapter->getInstantiable(substr($fqn, 1));
181  if($super !== null && !($object instanceof $super)){
182  throw new InvalidNodeException("$fqn does not extend/implement $super", $node);
183  }
184  return $object;
185  }
186 
187  $args = "";
188  if(($pos = mb_strpos($fqn, ":")) !== false){
189  $args = mb_substr($fqn, $pos + 1);
190  $fqn = mb_substr($fqn, 0, $pos);
191  }
192 
193  if($fqn{0} === "\\"){
194  $class = substr($fqn, 1);
195  }elseif($fqn{0} === "!"){
196  $class = libkinetic::getNamespace() . "\\Defaults\\" . substr($fqn, 1);
197  }else{
198  $class = $this->parser->getRoot()->asRootComponent()->getNamespace() . $fqn;
199  }
200 
201  if(!class_exists($class)){
202  throw new InvalidNodeException("Class \"$class\" ($fqn) does not exist", $node);
203  }
204 
205  if($super !== null && !is_subclass_of($class, $super)){
206  throw new InvalidNodeException("Class \"$class\" ($fqn) does not extend/implement $super", $node);
207  }
208 
209  $class = new ReflectionClass($class);
210  if(!$class->isInstantiable()){
211  throw new InvalidNodeException("Class \"$class\" ($fqn) is not instantiable", $node);
212  }
213  $constructor = $class->getConstructor();
214  if($constructor === null || $constructor->getNumberOfParameters() === 0){
215  return $class->newInstance();
216  }

219  throw new InvalidNodeException("Class must only accept one required parameter", $node);

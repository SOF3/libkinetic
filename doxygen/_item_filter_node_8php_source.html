53  public function ifn_setAttribute(string $name, string $value) : bool{
54  if($name === "ITEM" . "ID"){
55  $this->ifn_aggregate = false;
56  if(!empty($this->ifn_fromConfig)){ // returns true for false
57  throw $this->t_throw("Either declare \"fromConfig\" or \"itemId\"+\"itemDamage\", not both");
58  }
59  $this->ifn_fromConfig = false;
60  $this->ifn_id = $this->t_typeFix()->parseInt($value);
61  return true;
62  }
63 
64  if($name === "ITEM" . "DAMAGE"){
65  $this->ifn_aggregate = false;
66  if(!empty($this->ifn_fromConfig)){ // returns true for false
67  throw $this->t_throw("Either declare \"fromConfig\" or \"itemId\"+\"itemDamage\", not both");
68  }
69  $this->ifn_fromConfig = false;
70  if($value === "*" || $value === "-1" || strtoupper($value) === "ANY"){
71  $this->ifn_damage = -1;
72  }else{
73  $this->ifn_damage = $this->t_typeFix()->parseInt($value);
74  if($this->ifn_damage < 0){
75  $this->t_throw("\"itemDamage\" must either be \"*\" or a non-negative damage value");
76  }
77  }
78  return true;
79  }
80 
81  if($name === "FROM" . "CONFIG"){
82  $this->ifn_aggregate = false;
83  if(isset($this->ifn_fromConfig) && !$this->ifn_fromConfig){
84  throw $this->t_throw("Either declare \"fromConfig\" or \"itemId\"+\"itemDamage\", not both");
85  }
86  $this->ifn_fromConfig = true;
87  $this->ifn_configKey = $value;

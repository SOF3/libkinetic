160  public function resolveClass(KineticNode $node, ?string $fqn, ?string $super) : ?object{
161  if($fqn === null){
162  return null;
163  }
164 
165  if($fqn{0} === '$'){
166  $object = $this->adapter->getInstantiable(substr($fqn, 1));
167  if($super !== null && !($object instanceof $super)){
168  throw new InvalidNodeException("$fqn does not extend/implement $super", $node);
169  }
170  return $object;
171  }
172 
173  $args = "";
174  if(($pos = mb_strpos($fqn, ":")) !== false){
175  $args = mb_substr($fqn, $pos + 1);
176  $fqn = mb_substr($fqn, 0, $pos);
177  }
178 
179  if($fqn{0} === "\\"){
180  $class = substr($fqn, 1);
181  }elseif($fqn{0} === "!"){
182  $class = libkinetic::getNamespace() . "\\Defaults\\" . substr($fqn, 1);
183  }else{
184  $class = $this->parser->getRoot()->asRoot()->getNamespace() . $fqn;
185  }
186 
187  if(!class_exists($class)){
188  throw new InvalidNodeException("Class \"$class\" ($fqn) does not exist", $node);
189  }
190 
191  if($super !== null && !is_subclass_of($class, $super)){
192  throw new InvalidNodeException("Class \"$class\" ($fqn) does not extend/implement $super", $node);
193  }
194 
195  $class = new ReflectionClass($class);
196  if(!$class->isInstantiable()){
197  throw new InvalidNodeException("Class \"$class\" ($fqn) is not instantiable", $node);
198  }
199  $constructor = $class->getConstructor();
200  if($constructor === null || $constructor->getNumberOfParameters() === 0){
201  return $class->newInstance();
202  }
203 
204  if($constructor->getNumberOfRequiredParameters() > 1){
205  throw new InvalidNodeException("Class must only accept one required parameter", $node);
206  }
207 
208  $param = $constructor->getParameters()[0];
209  if($param->getClass() === null){
210  return $class->newInstance($args);
211  }
212 
213  if($param->getClass()->isInstance($this)){
214  return $class->newInstance($this, $args);
215  }
216 
217  if($param->getClass()->isInstance($this->plugin)){
218  return $class->newInstance($this->plugin, $args);
219  }
220 
221  return $class->newInstance($args);
222  }

251  public function handleFormResponse(Player $player, int $formId, $response) : void{
252  if(!isset($this->forms[$formId])){
253  return;
254  }
255 
256  $form = $this->forms[$formId];
257  if($player !== $form->target){
258  $this->plugin->getLogger()->error("{$player->getName()} tried respond to a form that was sent to {$form->target->getName()}. There is 99.99999999999999% chance that the plugin is outdated, shoghi needs to be blamed or the player is hacking poorly.");
259  return;
260  }
261  unset($this->forms[$formId]);
262 
263  try{
264  $result = ($form->onReceive)($response, $player);
265  if($result === true){
266  $this->sendForm($player, $form->formData, $form->onReceive);
267  }
268  }catch(ResendFormException $ex){
269  $this->forms[$formId] = $form;
270  $form->sendPacket();
271  }catch(TypeError|InvalidFormResponseException $error){
272  throw new RuntimeException("{$player->getName()} responded to a form with invalid data. Is {$this->plugin->getName()} using an outdated version of libkinetic?", 0, $error);
273  }

45  public function setAttribute(string $name, string $value) : bool{

103  private function parseConfig($config) : void{
104  $config = explode(":", (string) $config, 2);
105  if(is_numeric($config[0])){
106  $this->itemId = (int) $config[0];
107  }else{
108  $this->itemId = ItemFactory::fromString($config[0])->getId();
109  }
110  $this->itemDamage = in_array(strtoupper($config[1]) ?? "-1", ["-1", "*", "ANY"], true) ? -1 : (int) $config[1];
111  }

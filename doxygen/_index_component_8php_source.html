52  protected function onClickImpl(WindowRequest $request) : void{
54  $list = [];

63  }

69  $buttons = [];
70  foreach($list as $clickable){
71  $buttons[] = [
72  "text" => $request->translate($clickable->asClickableComponent()->getIndexName()),
73  ]; // TODO add icon
74  }

76  "type" => "form",

92  $cont = $this->node->findRoot()->asRootComponent()->getContCmd();

107  }
108 
109  foreach($list as $component){

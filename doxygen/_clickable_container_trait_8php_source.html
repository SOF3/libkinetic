41  protected function onClickImpl(WindowRequest $request) : void{
42  $user = $request->getUser();
43  if($user instanceof Player){
44  $this->onClickForm($request, $user);
45  }else{
46  $this->onClickCommandHeader($request);
47 
48  $subCmds = $this->getSubCommands($user);
49 
50  if(!empty($subCmds)){
51  if($this->getManager()->hasCont()){
52  $command = "/{$this->getManager()->getContComponents()[0]->getName()}";
53 
54  $this->getManager()->setContAction($user, function(array $args) use ($subCmds, $user, $command){
55  $result = $this->handleSubCommand($user, $args);
56 
57  if(!$result){
58  $user->sendMessage("Usage:");
59  self::sendSubCommands($user, $command, $subCmds);
60  }
61  });
62  }else{
63  $path = [];
64  for($node = $this->getNode(); $node !== null && !self::hasCommand($node, $user); $node = $node->nodeParent){
65  if($node->nodeParent->hasComponent(IndexComponent::class)){
66  $path[] = $node->asClickableComponent()->getArgName();
67  }else{
68  $request->send(libkinetic::MESSAGE_RUN_IN_GAME_FOR_INDEX);
69  goto after_subcommands;
70  }
71  }
72

74  goto after_subcommands;
75  }
76 
77  $command = "/{$node->asDirectEntryClickableComponent()->getCommands()[0]->getName()} " . implode(" ", array_reverse($path));
78  }
79 
80  self::sendSubCommands($user, $command, $subCmds);
81  }
82

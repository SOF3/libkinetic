124  public function resolveClass(KineticNode $node, string $fqn, ?string $super) : object{
125  if($fqn{0} === '$'){
126  $object = $this->adapter->getInstantiable(substr($fqn, 1));
127  if($super !== null && !($object instanceof $super)){
128  throw new InvalidNodeException("$fqn does not extend/implement $super", $node);
129  }
130  return $object;
131  }
132 
133  if($fqn{0} === "\\"){
134  $class = substr($fqn, 1);
135  }elseif($fqn{0} === "!"){
136  $class = libkinetic::getNamespace() . "\\" . substr($fqn, 1);
137  }else{
138  $class = $this->parser->getRoot()->getNamespace() . $fqn;
139  }
140 
141  if(!class_exists($class)){
142  throw new InvalidNodeException("Class \"$class\" ($fqn) does not exist", $node);
143  }
144 
145  if($super !== null && !is_subclass_of($class, $super)){
146  throw new InvalidNodeException("Class \"$class\" ($fqn) does not extend/implement $super", $node);
147  }
148 
149  $class = new ReflectionClass($class);
150  if(!$class->isInstantiable()){
151  throw new InvalidNodeException("Class \"$class\" ($fqn) is not instantiable", $node);
152  }
153  $constructor = $class->getConstructor();
154  if($constructor === null || $constructor->getNumberOfParameters() === 0){
155  return $class->newInstance();
156  }
157 
158  if($constructor->getNumberOfRequiredParameters() > 1){
159  throw new InvalidNodeException("Class must only accept one required parameter", $node);
160  }
161 
162  $param = $constructor->getParameters()[0];
163  if($param->getClass() === null || !$param->getClass()->isInstance($this->plugin)){
164  return $class->newInstance();
165  }
166 
167  return $class->newInstance($this->plugin);
168  }
